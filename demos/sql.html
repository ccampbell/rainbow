Simple MSSQL example: 

/* MSSQL example */
USE AdventureWorks2012;
GO
SELECT Name, ProductNumber, ListPrice AS Price
FROM Production.Product 
WHERE ProductLine = "XYZ"
AND DaysToManufacture < 4
ORDER BY Name ASC;
